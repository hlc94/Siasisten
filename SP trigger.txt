CREATE TRIGGER tambah_lowongan
AFTER INSERT
ON lowongan
FOR EACH ROW
EXECUTE PROCEDURE tambah_lowongan();

CREATE OR REPLACE FUNCTION tambah_lowongan
	RETURNS "trigger" AS $$
	DECLARE
	BEGIN
	UPDATE lowongan set id_st_lamaran = id_st_lamaran + NEW.id_st_lamaran
	WHERE idlowongan = NEW.baruidlowongan;
	RETURN NEW;
END;
$$ LANGUAGE plpgsql;
